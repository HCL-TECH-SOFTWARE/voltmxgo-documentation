 * @example rosetta.tan(.5); // returns 0.5463024898437905

 */
function tan(...angle) {
    const list = angle.map(n => n instanceof Error ? n : API.isDefaultOpenFormula() ? openf.TAN(n) : notesf.Tan(n));
